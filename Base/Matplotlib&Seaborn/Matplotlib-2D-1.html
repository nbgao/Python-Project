6. 等高线图 contourf¶

In [49]:
x = np.linspace(-5,5,500)
y = np.linspace(-5,5,500)

X, Y = np.meshgrid(x,y)
Z = (1-X/2+X**3+Y**4) * np.exp(-X**2-Y**2)

plt.figure(figsize=(8,8))
plt.contourf(X,Y,Z)
plt.show()
